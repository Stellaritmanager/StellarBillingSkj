ALTER TRIGGER [dbo].[trigger_SHbilldetails_update]
   ON  [dbo].[SHbilldetails]
   AFTER Update
AS 
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	declare @count int;
	SET NOCOUNT ON;
 
    update SHRackPartionProduct set SHRackPartionProduct.noofitems=convert(int,SHRackPartionProduct.noofitems)-(convert(int,inserted.quantity)-convert(int,deleted.quantity))
    from
    SHRackPartionProduct,
	inserted,
	deleted
	where
	  SHRackPartionProduct.ProductID=inserted.ProductID
	  and inserted.ProductID=deleted.ProductID
     
END
